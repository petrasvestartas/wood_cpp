14 double length(double x, double y, double z)
15 {
16
17 double len;
18 x = fabs(x);
19 y = fabs(y);
20 z = fabs(z);
21 if (y >= x && y >= z)
22 {
23 len = x;
24 x = y;
25 y = len;
26 }
27 else if (z >= x && z >= y)
28 {
29 len = x;
30 x = z;
31 z = len;
32 }
33
34 // 15 September 2003 Dale Lear
35 // For small denormalized doubles (positive but smaller
36 // than DBL_MIN), some compilers/FPUs set 1.0/x to +INF.
37 // Without the ON_DBL_MIN test we end up with
38 // microscopic vectors that have infinte length!
39 //
40 // This code is absolutely necessary. It is a critical
41 // part of the bug fix for RR 11217.
42 if (x > ON_DBL_MIN)
43 {
44 y /= x;
45 z /= x;
46 len = x * sqrt(1.0 + y * y + z * z);
47 }
48 else if (x > 0.0 && ON_IS_FINITE(x))
49 len = x;
50 else
51 len = 0.0;
52
53 return len;
54 }

56 bool unitize(IK::Vector_3 &vector)
57 {
58 bool rc = false;
59 // Since x,y,z are floats, d will not be denormalized and the
60 // ON_DBL_MIN tests in ON_2dVector::Unitize() are not needed.
61
62 double d = length(vector.hx(), vector.hy(), vector.hz());
63 if (d > 0.0)
64 {
65 double dx = vector.hx();
66 double dy = vector.hy();
67 double dz = vector.hz();
68 vector = IK::Vector_3(
69 (dx / d),
70 (dy / d),
71 (dz / d));
72 rc = true;
73 }
74 return rc;
75 }

77 int is_parallel_to(const IK::Vector_3 &v0, const IK::Vector_3 &v1)
78 {
79 int rc = 0;
80
81 double ll = length(v0.x(), v0.y(), v0.z()) * length(v1.x(), v1.y(), v1.z());
82 if (ll > 0.0)
83 {
84 const double cos_angle = (v0.x() * v1.x() + v0.y() * v1.y() + v0.z() * v1.z()) / ll;
85 const double cos_tol = cos(wood::globals::ANGLE);
86 if (cos_angle >= cos_tol)
87 rc = 1;
88 else if (cos_angle <= -cos_tol)
89 rc = -1;
90 }
91 return rc;
92 }
