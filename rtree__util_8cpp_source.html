18 IK::Point_3 point_at(const IK::Segment_3 &line, const double &t)
19 {
20 const double s = 1.0 - t;
21
22 return IK::Point_3(
23 (line[0].x() == line[1].x()) ? line[0].x() : s * line[0].x() + t * line[1].x(),
24 (line[0].y() == line[1].y()) ? line[0].y() : s * line[0].y() + t * line[1].y(),
25 (line[0].z() == line[1].z()) ? line[0].z() : s * line[0].z() + t * line[1].z());
26 }

28 bool closest_point_to(const IK::Point_3 &point, const IK::Segment_3 &line, double &t)
29 {
30 bool rc = false;
31
32 const IK::Vector_3 D = line.to_vector();
33 const double DoD = D.squared_length();
34
35 if (DoD > 0.0)
36 {
37 if ((point - line[0]).squared_length() <= (point - line[1]).squared_length())
38 t = ((point - line[0]) * D) / DoD;
39 else
40 t = 1.0 + ((point - line[1]) * D) / DoD;
41
42 rc = true;
43 }
44 else
45 { // (GBA) Closest point to a degenerate line works as well
46 t = 0.0;
47 rc = true;
48 }
49
50 return rc;
51 }

53 double get_closest_distance(const IK::Point_3 &point, CGAL_Polyline &polyline, size_t &edge_id)
54 {
55 edge_id = 0;
56 IK::Segment_3 segment(polyline[0], polyline[1]);
57 double closest_distance = DBL_MAX;
58
59 for (size_t i = 0; i < polyline.size() - 1; i++)
60 {
61 IK::Segment_3 segment_(polyline[i], polyline[i + 1]);
62
63 double t;
64 closest_point_to(point, segment_, t);
65
66 double closest_distance_temp = std::abs(CGAL::squared_distance(point, point_at(segment_, t)));
67 if (closest_distance_temp < closest_distance_temp)
68 {
69 closest_distance = closest_distance_temp;
70 edge_id = i;
71 }
72
73 if (closest_distance < wood::globals::DISTANCE_SQUARED)
74 break;
75 }
76
77 return closest_distance;
78 }
