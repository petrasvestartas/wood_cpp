18 double length(double x, double y, double z)
19 {
20
21 double len;
22 x = fabs(x);
23 y = fabs(y);
24 z = fabs(z);
25 if (y >= x && y >= z)
26 {
27 len = x;
28 x = y;
29 y = len;
30 }
31 else if (z >= x && z >= y)
32 {
33 len = x;
34 x = z;
35 z = len;
36 }
37
38 // 15 September 2003 Dale Lear
39 // For small denormalized doubles (positive but smaller
40 // than DBL_MIN), some compilers/FPUs set 1.0/x to +INF.
41 // Without the ON_DBL_MIN test we end up with
42 // microscopic vectors that have infinte length!
43 //
44 // This code is absolutely necessary. It is a critical
45 // part of the bug fix for RR 11217.
46 if (x > ON_DBL_MIN)
47 {
48 y /= x;
49 z /= x;
50 len = x * sqrt(1.0 + y * y + z * z);
51 }
52 else if (x > 0.0 && ON_IS_FINITE(x))
53 len = x;
54 else
55 len = 0.0;
56
57 return len;
58 }

60 bool unitize(IK::Vector_3 &vector)
61 {
62 bool rc = false;
63 // Since x,y,z are floats, d will not be denormalized and the
64 // ON_DBL_MIN tests in ON_2dVector::Unitize() are not needed.
65
66 double d = length(vector.hx(), vector.hy(), vector.hz());
67 if (d > 0.0)
68 {
69 double dx = vector.hx();
70 double dy = vector.hy();
71 double dz = vector.hz();
72 vector = IK::Vector_3(
73 (dx / d),
74 (dy / d),
75 (dz / d));
76 rc = true;
77 }
78 return rc;
79 }

224 CGAL::Aff_transformation_3<IK> xy_to_plane(const IK::Vector_3 &origin, const IK::Vector_3 &x_axis, const IK::Vector_3 &y_axis, const IK::Vector_3 &z_axis)
225 {
226 // Vectors must be unitized, else the result is wrong
227 IK::Vector_3 _x_axis = x_axis;
228 IK::Vector_3 _y_axis = y_axis;
229 IK::Vector_3 _z_axis = z_axis;

233 internal::unitize(_z_axis);

236 CGAL::Aff_transformation_3<IK> f(
237 _x_axis.x(), _y_axis.x(), _z_axis.x(),
238 _x_axis.y(), _y_axis.y(), _z_axis.y(),
239 _x_axis.z(), _y_axis.z(), _z_axis.z());
240

242 CGAL::Aff_transformation_3<IK> t(CGAL::TRANSLATION, IK::Vector_3(origin.x(), origin.y(), origin.z()));

247 CGAL::Aff_transformation_3<IK> axis_rotation(const double &angle, const IK::Vector_3 &axis)
248 {
249 // create matrix of the rotation
250 IK::RT
251 c = cos(angle),
252 s = sin(angle),
253 ux(axis.x()),
254 uy(axis.y()),
255 uz(axis.z());
256
257 IK::RT matrix[12] =
258 {
259 ux * ux * (1 - c) + c, ux * uy * (1 - c) - uz * s, ux * uz * (1 - c) + uy * s, 0,
260 ux * uy * (1 - c) + uz * s, uy * uy * (1 - c) + c, uy * uz * (1 - c) - ux * s, 0,
261 ux * uz * (1 - c) - uy * s, uy * uz * (1 - c) + ux * s, uz * uz * (1 - c) + c, 0};
262
263 return CGAL::Aff_transformation_3<IK>(matrix[0], matrix[1], matrix[2],
264 matrix[3], matrix[4], matrix[5],
265 matrix[6], matrix[7], matrix[8],
266 matrix[9], matrix[10], matrix[11]);
267 }

271 std::tuple<IK::Point_3, IK::Plane_3, double> get_polylabel(const std::vector<CGAL_Polyline> &polylines, double precision)
272 {
273
275 // copy polylines for orienting from 3D to 2D
277 std::vector<CGAL_Polyline> polylines_copy = polylines;
278
280 // Create Transformation to XY plane
282 IK::Point_3 origin;
283 IK::Plane_3 plane;
284 cgal::polyline_util::get_fast_plane(polylines_copy[0], origin, plane);
285
286 CGAL::Aff_transformation_3<IK> xform_toXY = internal::plane_to_xy(origin, plane);
287 CGAL::Aff_transformation_3<IK> xform_toXY_Inv = xform_toXY.inverse();
288 for (auto &polyline : polylines_copy)
289 cgal::polyline_util::transform(polyline, xform_toXY);
290
292 // Sort polylines based on bounding-box to detect which polylines are holes, this only works if the polygon does not have holes within holes
294 std::vector<double> len = std::vector<double>(polylines_copy.size());
295 std::vector<int> ids(len.size());
296 std::iota(begin(ids), end(ids), 0);
297
299 // Get bounding box length of polylines_copy
301 if (len.size() > 0)
302 {
303 for (size_t i = 0; i < polylines_copy.size(); i++)
304 {
305 CGAL::Bbox_3 AABB = CGAL::bbox_3(polylines_copy[i].begin(), polylines_copy[i].end(), IK());
306 auto p0 = IK::Point_3(AABB.xmin(), AABB.ymin(), AABB.zmin());
307 auto p1 = IK::Point_3(AABB.xmax(), AABB.ymax(), AABB.zmax());
308 len[i] = CGAL::squared_distance(p0, p1);
309 }
311 // Sort polylines_copy indices by the bounding box length
312 // WARNING orientation is not checked, which should be fine with the Polylabel algorithm
314
315 std::vector<int> ids(len.size());
316 std::sort(begin(ids), end(ids), [&](int ia, int ib)
317 { return len[ia] < len[ib]; });
318 std::reverse(begin(ids), end(ids));
319 }
320
322 // Convert polylines_copy to Polylabel datastructure, be aware that the polyline considers all polygons closed, so skip the last point
324 mapbox::geometry::polygon<double> polygon_with_holes(polylines_copy.size());
325
326 for (size_t i = 0; i < polylines_copy.size(); i++)
327 {
328 for (size_t j = 0; j < polylines_copy[ids[i]].size() - 1; j++)
329 {
330 polygon_with_holes[i].emplace_back(polylines_copy[ids[i]][j].x(), polylines_copy[ids[i]][j].y());
331 // std::cout << polylines_copy[ids[i]][j].x() << " " << polylines_copy[ids[i]][j].y() << std::endl;
332 }
333 }
334
336 // Run the Polylabel algorithm, WARNING numbers are in 2D, first two doubles are center, the third is the radius
338 std::array<double, 3> center_and_radius = mapbox::polylabel(polygon_with_holes, 1.0);
339
341 // Convert from the Polylabel datastructure to CGAL
343 IK::Point_3 center(center_and_radius[0], center_and_radius[1], 0);
344 center = center.transform(xform_toXY_Inv);
345
347 // Output
349 return std::tuple<IK::Point_3, IK::Plane_3, double>(center, plane, center_and_radius[2]);
350 }

352 void get_polylabel_circle_division_points(const IK::Vector_3 &division_direction_in_3d, const std::vector<CGAL_Polyline> &polylines, std::vector<IK::Point_3> &points, int division, double scale, double precision, bool orient_to_closest_edge)
353 {
354 // run the polylabel algorithm
355 std::tuple<IK::Point_3, IK::Plane_3, double> circle = get_polylabel(polylines, precision);
356 IK::Vector_3 center(std::get<0>(circle).hx(), std::get<0>(circle).hy(), std::get<0>(circle).hz());
357 bool is_direction_valid = division_direction_in_3d.hx() != 0.0 && division_direction_in_3d.hy() != 0.0 && division_direction_in_3d.hz() != 0.0;
358
360 // Find closest edge direction to the center
362
363 IK::Point_3 center_point(std::get<0>(circle).hx(), std::get<0>(circle).hy(), std::get<0>(circle).hz());
364 double distance_from_center_to_polyline_edge = DBL_MAX;
365 size_t edge_id0 = 0;
366 size_t edge_id1 = 0;
367 if (orient_to_closest_edge)
368 {
369
370 for (size_t i = 0; i < polylines.size(); i++)
371 {
372 for (size_t j = 0; j < polylines[i].size() - 1; j++)
373 {
374
375 IK::Segment_3 polyline_segment(polylines[i][j], polylines[i][j + 1]);
376 IK::Point_3 projection = polyline_segment.supporting_line().projection(center_point);
377
378 // closest point paraneters
379 bool has_on = false;
380 const IK::Vector_3 polyline_segment_vector = polyline_segment.to_vector();
381 const double polyline_segment_vector_squared_len = polyline_segment_vector.squared_length();
382 double t = 0;
383
384 if (polyline_segment_vector.squared_length() > 0.0)
385 {
386 if ((projection - polyline_segment[0]).squared_length() <= (projection - polyline_segment[1]).squared_length())
387 t = ((projection - polyline_segment[0]) * polyline_segment_vector) / polyline_segment_vector_squared_len;
388 else
389 t = 1.0 + ((projection - polyline_segment[1]) * polyline_segment_vector) / polyline_segment_vector_squared_len;
390
391 has_on = t >= 0 && t <= 1;
392 }
393 else
394 { // (GBA) Closest point to a degenerate line works as well
395 t = 0.0;
396 has_on = false;
397 }
398
399 // check if the point is on the edge
400 if (has_on)
401 {
402 double temp_distance_from_center_to_polyline_edge = CGAL::squared_distance(center_point, projection);
403 if (temp_distance_from_center_to_polyline_edge < distance_from_center_to_polyline_edge)
404 {
405 distance_from_center_to_polyline_edge = temp_distance_from_center_to_polyline_edge;
406 edge_id0 = i;
407 edge_id1 = j;
408 }
409 }
410 }
411 }
412 }
413
414 // orientation direction
415 IK::Vector_3 division_direction_in_3d_ = orient_to_closest_edge ? polylines[edge_id0][edge_id1 + 1] - polylines[edge_id0][edge_id1 + 0] : IK::Vector_3(division_direction_in_3d.hx(), division_direction_in_3d.hy(), division_direction_in_3d.hz());
416
417 // plane vectors for orienting again to 2d for circle points dvisions
418 IK::Vector_3 x_axis = is_direction_valid || orient_to_closest_edge ? division_direction_in_3d_ : std::get<1>(circle).base1();
419 IK::Vector_3 y_axis = is_direction_valid || orient_to_closest_edge ? CGAL::cross_product(division_direction_in_3d_, std::get<1>(circle).orthogonal_vector()) : std::get<1>(circle).base2();
420 IK::Vector_3 z_axis = std::get<1>(circle).orthogonal_vector();
421
422 // run the division method which is the output -> points
423 internal::circle_points(center, x_axis, y_axis, z_axis, points, division, std::get<2>(circle) * scale);
424 }

426 bool inscribe_rectangle_in_convex_polygon(const std::vector<CGAL_Polyline> &polylines, CGAL_Polyline &result, std::vector<IK::Point_3> &points, IK::Segment_3 &direction, double scale, double precision, double rectangle_division_distance)
427 {
428
430 // copy polylines
432 std::vector<CGAL_Polyline> polylines_copy = polylines;
433
435 // user direction
437
438 IK::Point_3 o;
439 IK::Vector_3 x, y, z;
440 if (direction.squared_length() > wood::globals::DISTANCE_SQUARED)
441 {
442
443 internal::get_fast_plane(polylines_copy[0], o, x, y, z);
444 x = direction.to_vector();
445 y = CGAL::cross_product(x, z);
446 }
447 else
448 {
449 // this was never checked
450 internal::get_average_plane_axes_oriented_to_longest_edge(polylines_copy[0], o, x, y, z);
451 }
452
453 // if (polylines_copy[0].size() != 5)
454 // return true;
455
457 // Create Transformation to XY plane
459 CGAL::Aff_transformation_3<IK>
460 xform_toXY = internal::plane_to_xy(o, x, y, z);
461
462 // without rotation the algorithm does not work, is it a bug of CGAL?
463 CGAL::Aff_transformation_3<IK> rot = internal::axis_rotation(0.0001, IK::Vector_3(0, 0, 1));
464 xform_toXY = rot * xform_toXY;
465
466 CGAL::Aff_transformation_3<IK>
467 xform_toXY_Inv = xform_toXY.inverse();
468
470 // Count how many points we have
472 size_t point_count = 0;
473
474 for (auto &polyline : polylines_copy)
475 point_count += polyline.size();
476
478 // Compute 2D bounding box to get a division step, the division step is equal to = diagonal_length/(20*precision)
480 std::vector<IK::Point_2> points_2d;
481 points_2d.reserve(point_count);
482
483 for (auto &polyline : polylines_copy)
484 {
485 cgal::polyline_util::transform(polyline, xform_toXY);
486 for (auto &p : polyline)
487 points_2d.push_back(IK::Point_2(p.hx(), p.hy()));
488 points_2d.pop_back();
489 }
490
491 CGAL::Bbox_2 bbox = CGAL::bbox_2(points_2d.begin(), points_2d.end(), IK());
492 auto diagonal = {IK::Point_3(bbox.xmin(), bbox.ymin(), 0), IK::Point_3(bbox.xmax(), bbox.ymax(), 0)};
493 double squared_diagonal_length = CGAL::squared_distance(IK::Point_3(bbox.xmin(), bbox.ymin(), 0), IK::Point_3(bbox.xmax(), bbox.ymax(), 0));
494 double step_division = std::sqrt(squared_diagonal_length) / (50.0 * precision);
495
497 // Get iso rectangle as the first input for the algorithm
499 IK::Iso_rectangle_2 bbox_rectangle(bbox);
500 CGAL::Largest_empty_iso_rectangle_2<IK> empty_rect(bbox_rectangle);
501
503 // Interpolate edge points in by the precomputed division step
505
506 CGAL_Polyline division_points;
507 CGAL_Polyline division_points_temp;
508 for (auto &polyline : polylines_copy)
509 for (int i = 0; i < polyline.size() - 1; i++)
510 {
511 division_points.clear();
512 int divisions = (int)std::min(100.0, std::sqrt(CGAL::squared_distance(polyline[i], polyline[i + 1])) / step_division);
513 cgal::vector_util::interpolate_points(polyline[i], polyline[i + 1], divisions, division_points, 2);
514 division_points_temp.insert(division_points_temp.end(), division_points.begin(), division_points.end());
515
516 for (auto &point : division_points)
517 {
518 empty_rect.push_back(IK::Point_2(point.hx(), point.hy()));
519 }
520 }
521
523 // Get the largest empty iso rectangle
525 IK::Iso_rectangle_2 rectangle = empty_rect.get_largest_empty_iso_rectangle();
526 double x_dist = (1 - scale) * std::abs(rectangle.max().hx() - rectangle.min().hx());
527 double y_dist = (1 - scale) * std::abs(rectangle.max().hy() - rectangle.min().hy());
528 x_dist = std::min(x_dist, y_dist);
529 y_dist = x_dist;
530
531 CGAL_Polyline rect = {
532 IK::Point_3(rectangle.min().hx() + x_dist, rectangle.min().hy() + y_dist, 0),
533 IK::Point_3(rectangle.min().hx() + x_dist, rectangle.max().hy() - y_dist, 0),
534 IK::Point_3(rectangle.max().hx() - x_dist, rectangle.max().hy() - y_dist, 0),
535 IK::Point_3(rectangle.max().hx() - x_dist, rectangle.min().hy() + y_dist, 0),
536 IK::Point_3(rectangle.min().hx() + x_dist, rectangle.min().hy() + y_dist, 0),
537 };
538
540 // Orient Back to 3D
542 for (auto &point : rect)
543 point = point.transform(xform_toXY_Inv);
544
546 // point on the edge of the rectangle or division of points inside the rectangle
548 if (rectangle_division_distance > wood::globals::DISTANCE)
549 {
551 // divide edges by distance
553 for (int i = 0; i < rect.size() - 1; i++)
554 {
555 division_points.clear();
556 int divisions = (int)std::min(25.0, std::sqrt(CGAL::squared_distance(rect[i], rect[i + 1])) / rectangle_division_distance);
557 cgal::vector_util::interpolate_points(rect[i], rect[i + 1], divisions, division_points, 2);
558 points.insert(points.end(), division_points.begin(), division_points.end());
559 }
560 // cgal_polyline_util::transform(division_points_temp, xform_toXY_Inv);
561 // points.clear();
562 // points.insert(points.end(), division_points_temp.begin(), division_points_temp.end());
563 // auto p0_min = (IK::Point_3(bbox.xmin(), bbox.ymin(), 0)).transform(xform_toXY_Inv);
564 // auto p0_max = (IK::Point_3(bbox.xmax(), bbox.ymax(), 0)).transform(xform_toXY_Inv);
565 // points.emplace_back(p0_min);
566 // points.emplace_back(p0_max);
567 }
568 else if (rectangle_division_distance < -wood::globals::DISTANCE)
569 {
571 // get divisions of each edge and take the smallest division
573 double unsigned_rectangle_division_distance = std::abs(rectangle_division_distance);
574 double edge_dist_00 = std::sqrt(CGAL::squared_distance(rect[0], rect[1]));
575 double edge_dist_01 = std::sqrt(CGAL::squared_distance(rect[3], rect[2]));
576 double edge_dist_10 = std::sqrt(CGAL::squared_distance(rect[1], rect[2]));
577 double edge_dist_11 = std::sqrt(CGAL::squared_distance(rect[0], rect[3]));
578
579 int divisions_u = (int)(std::min(std::ceil(edge_dist_00 / unsigned_rectangle_division_distance), std::ceil(edge_dist_01 / unsigned_rectangle_division_distance)));
580 int divisions_v = (int)(std::min(std::ceil(edge_dist_10 / unsigned_rectangle_division_distance), std::ceil(edge_dist_11 / unsigned_rectangle_division_distance)));
